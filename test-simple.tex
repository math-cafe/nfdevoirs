    \begin{partie}{Équations du second degré}

        \begin{exercice}{Application directe}

            Résolvez les équations suivantes :

            \begin{question}{points=3, niveau=2}
                $3x^2 - 2x + 1 = 0$
            \end{question}

            \begin{correction}
                Calculons le discriminant : $\Delta = b^2 - 4ac = 4 - 12 = -8 < 0$

                L'équation n'a pas de solution réelle.
            \end{correction}

            \begin{question}{points=3}
                $x^2 - 6x + 9 = 0$
            \end{question}

            \begin{correction}
                $\Delta = 36 - 36 = 0$

                Une solution double : $x_0 = \frac{6}{2} = 3$
            \end{correction}

        \end{exercice}

        \begin{exercice}{Exercice Bonus}

            Cet exercice est entièrement bonus.

            \begin{question}{bonus=5, niveau=5}
                Montrez que $(2\sqrt{3} + 1)^2 = 13 + 4\sqrt{3}$
            \end{question}

            \begin{correction}
                $(2\sqrt{3} + 1)^2 = 4 \times 3 + 2 \times 2\sqrt{3} + 1 = 12 + 4\sqrt{3} + 1 = 13 + 4\sqrt{3}$
            \end{correction}

        \end{exercice}

    \end{partie}

    \begin{partie}{Questions à choix multiples}

        \begin{exercice}{QCM - Style case}

            \begin{qcm}{points=2, style=case, col=1, niveau=1}
                Combien vaut $2^3$ ?
                \begin{choix}
                    \proposition{6}
                    \proposition*{8}
                    \proposition{9}
                    \proposition{12}
                \end{choix}
            \end{qcm}

            \begin{correction}
                $2^3 = 2 \times 2 \times 2 = 8$
            \end{correction}

            \begin{qcm}{points=3, style=case, col=2, niveau=3}
                Quels sont les nombres pairs parmi les suivants ?
                \begin{choix}
                    \proposition{1}
                    \proposition*{2}
                    \proposition{3}
                    \proposition*{4}
                    \proposition{5}
                    \proposition*{6}
                \end{choix}
            \end{qcm}

            \begin{correction}
                Un nombre pair est divisible par 2. Les réponses sont donc 2, 4 et 6.
            \end{correction}

        \end{exercice}

        \begin{exercice}{QCM - Style alphabétique}

            \begin{qcm}{points=2, bonus=1, style=alpha, col=1, niveau=4}
                Quelle est la dérivée de $f(x) = x^2$ ?
                \begin{choix}
                    \proposition{$2x^2$}
                    \proposition*{$2x$}
                    \proposition{$x$}
                    \proposition{$\frac{x^3}{3}$}
                \end{choix}
            \end{qcm}

            \begin{correction}
                La dérivée de $x^n$ est $nx^{n-1}$, donc la dérivée de $x^2$ est $2x$.
            \end{correction}

            \begin{qcm}{points=4, style=alpha, col=3, niveau=5}
                Dans un triangle rectangle, quelles relations sont vraies ? (théorème de Pythagore : $a^2 + b^2 = c^2$)
                \begin{choix}
                    \proposition*{$\sin^2(\theta) + \cos^2(\theta) = 1$}
                    \proposition{$\tan(\theta) = \frac{\sin(\theta)}{\cos(\theta)}$ seulement si $\theta = 45°$}
                    \proposition*{$\tan(\theta) = \frac{\sin(\theta)}{\cos(\theta)}$}
                    \proposition{$\sin(\theta) = \cos(\theta)$ pour tout $\theta$}
                    \proposition*{L'hypoténuse est le côté le plus long}
                    \proposition{Les angles aigus sont toujours égaux}
                \end{choix}
            \end{qcm}

            \begin{correction}
                Les relations trigonométriques fondamentales sont toujours vraies dans un triangle rectangle.
            \end{correction}

        \end{exercice}

        \begin{exercice}{QCM - Style mixte}

            \begin{qcm}{points=3, style=mix, col=2, niveau=2}
                Soit $f(x) = 3x + 2$. Que vaut $f(1)$ ?
                \begin{choix}
                    \proposition{2}
                    \proposition{3}
                    \proposition*{5}
                    \proposition{6}
                \end{choix}
            \end{qcm}

            \begin{correction}
                $f(1) = 3 \times 1 + 2 = 3 + 2 = 5$
            \end{correction}

            \begin{qcm}{points=2, style=mix, col=1}
                Mathématiques : quel(s) énoncé(s) est/sont correct(s) ?

                Soit $A = \{1, 2, 3\}$ et $B = \{2, 3, 4\}$.
                \begin{choix}
                    \proposition*{$A \cap B = \{2, 3\}$}
                    \proposition{$A \cup B = \{1, 4\}$}
                    \proposition*{$A \cup B = \{1, 2, 3, 4\}$}
                    \proposition{$A \cap B = \{1, 2, 3, 4\}$}
                \end{choix}
            \end{qcm}

            \begin{correction}
                $A \cap B$ (intersection) contient les éléments communs : $\{2, 3\}$.
                $A \cup B$ (union) contient tous les éléments : $\{1, 2, 3, 4\}$.
            \end{correction}

        \end{exercice}
